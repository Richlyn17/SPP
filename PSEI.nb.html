ARIMA

This is a command for auto arima.

bestfit_psei <- list(aicc=Inf)
for(i in 1:25)
{
  fit_psei <- auto.arima(psei, xreg=fourier(psei, K=i), seasonal=FALSE)
  if(fit_psei$aicc < bestfit_psei$aicc)
    bestfit_psei <- fit_psei
  else break;
}
