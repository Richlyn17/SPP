SPP for PLC

library(forecast)
library(ggplot2)
library(readr)
PLC <- read_csv("~/R/SPP/CSV/PLC.csv", col_types = cols(Week = col_date(format = "%Y-%m-%d")))
plc <- ts(PLC[,2], freq=365.25/7, start=2015+51/365.25)
plc %>%
  autoplot()

plot(decompose(plc))

training_plc <- subset(plc, end=length(plc)-3)
test_plc <- subset(plc, start=length(plc)-2)
fc_training_plc<-forecast(training_plc)
fc_training_plc %>%
  forecast(h=26) %>%
  autoplot() +autolayer(test_plc)

fc_training_plc
fc_plc<-forecast(plc)
fc_plc %>%
forecast(h=26) %>%
  autoplot()
